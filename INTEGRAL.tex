			\addplot+[no marks,red,domain=-10:10,samples=150] (3, x);
			\addlegendentry{$x = 3$}

	\\ Therefore the area of a raegion under the curve is :
	\begin{eqnarray*}
		\int_{-1}^{3} -x^2+2x+8\; dx &=& -\frac{x^3}{3} + x^2 + 8x \Big|_{-1}^{3} \\
		&=& 8 + 32 - \frac{28}{3} \\
		&=& \left(-\frac{27}{3} + 9 + 24\right) - \left(-\frac{1}{3}+1-8\right)\\
		&=& \frac{92}{3} \; \textrm{unit area}
	\end{eqnarray*}
